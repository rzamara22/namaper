<body>

  <input type="text" id="search" placeholder="Nama Instansi / Perusahaan..." oninput="filter(this.value)">

  <ul id="results"></ul>


  <script>

    let items = [];

    fetch("https://script.google.com/macros/s/AKfycbwg3TX_oEplFqrgzE-d-Cn0uCHW0KcN-2o8qKJed5CsHmLPuULeJ2pHCCDbRyuj3qQy/exec")

      .then(res => res.json())

      .then(data => {

        items = data;

      });


    function filter(text) {

      let list = document.getElementById("results");

      list.innerHTML = "";

      items

        .filter(item => item.toLowerCase().includes(text.toLowerCase()))

        .slice(0, 10) // batasi 10 hasil

        .forEach(item => {

          let li = document.createElement("li");

          li.textContent = item;

          list.appendChild(li);

        });

    }

  </script>
